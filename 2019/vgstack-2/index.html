# vgstack/games/models.py

import uuid

from django.conf import settings
from django.db import models
from django.utils.translation import gettext as _

from systems.models import System


class Game(models.Model):
    STATUS_CHOICES = (
        ("UNBEATEN", "unbeaten"),
        ("BEATEN", "beaten"),
        ("COMPLETED", "completed"),
    )
    REGIONS = (
        ("NA", "North America"),
        ("PAL", "PAL"),
        ("JPN", "Japan"),
        ("CHN", "China"),
        ("KOR", "Korea"),
        ("BRZ", "Brazil"),
    )

    id = models.UUIDField(primary_key=True, default=uuid.uuid4, editable=False)
    name = models.CharField(max_length=100)
    status = models.CharField(
        max_length=25, choices=STATUS_CHOICES, null=True, blank=True, default=None
    )
    region = models.CharField(max_length=10, choices=REGIONS, default="NA")
    notes = models.TextField(null=True, blank=True)

    owner = models.ForeignKey(
        settings.AUTH_USER_MODEL,
        on_delete=models.CASCADE,
        related_name="games",
        related_query_name="game",
    )

    system = models.ForeignKey(
        System,
        on_delete=models.PROTECT,
        related_name="systems",
        related_query_name="system",
    )

    def __str__(self):
        return self.name

    class Meta:
        indexes = [models.Index(fields=["status", "region"])]
        ordering = ["-name"]
        verbose_name = _("game")
        verbose_name_plural = _("games")

# vgstack/systems/models.py

import uuid

from django.db import models
from django.utils.translation import gettext as _

from manufacturers.models import Manufacturer


class System(models.Model):
    id = models.UUIDField(primary_key=True, default=uuid.uuid4, editable=False)
    name = models.CharField(max_length=50)
    release_year = models.PositiveIntegerField()
    manufacturer = models.ForeignKey(Manufacturer, on_delete=models.CASCADE)

    def __str__(self):
        return f"{self.manufacturer} {self.name}"

    class Meta:
        ordering = ["release_year", "-name"]
        verbose_name = _("system")
        verbose_name_plural = _("systems")

from django.utils.translation import gettext as _

        return self.name

    class Meta:
        indexes = [models.Index(fields=["name"])]

        verbose_name = _("manufacturer")
        verbose_name_plural = _("manufacturers")

# vgstack/games/tests.py

from rest_framework import status
from rest_framework.reverse import reverse
from rest_framework.test import APITestCase

from games.factories import GameFactory
from games.models import Game
from systems.factories import SystemFactory
from users.factories import CustomUserFactory


class GameModelTests(APITestCase):
    def test_can_save_and_get_game(self):
        Game.objects.create(
            name="test game", system=SystemFactory(), owner=CustomUserFactory()
        )
        game = Game.objects.filter(name="test game")
        self.assertEquals(game.count(), 1)

    def test_game_str_representation(self):
        game = Game.objects.create(
            name="test game", system=SystemFactory(), owner=CustomUserFactory()
        )
        self.assertEquals(str(game), "test game")

    def test_can_obtain_game_from_owner(self):
        owner = CustomUserFactory()
        system = SystemFactory()

        Game.objects.create(name="test game", system=system, owner=owner)

        game = owner.games.filter(owner=owner)
        self.assertEquals(game.count(), 1)

        Game.objects.create(name="second test game", system=system, owner=owner)

        games = owner.games.all()

        self.assertEquals(games.count(), 2)


class GameViewTests(APITestCase):
    @classmethod
    def setUpTestData(cls):
        cls.user = CustomUserFactory()

    def test_list_url_exists(self):
        response = self.client.get(reverse("game-list"))
        self.assertEquals(response.status_code, status.HTTP_200_OK)

    def test_retrieve_url_exists(self):
        game = GameFactory()
        response = self.client.get(reverse("game-detail", args=[game.pk]))
        self.assertEquals(response.status_code, status.HTTP_200_OK)

    def test_can_filter_by_owner(self):
        user_1 = CustomUserFactory()
        user_2 = CustomUserFactory()
        GameFactory(owner=user_1)
        GameFactory(owner=user_1)
        GameFactory(owner=user_1)
        GameFactory(owner=user_2)

        response = self.client.get(f"{reverse('game-list')}?owner={user_1.pk}")
        self.assertEquals(response.status_code, status.HTTP_200_OK)
        self.assertEquals(len(response.data), 3)
